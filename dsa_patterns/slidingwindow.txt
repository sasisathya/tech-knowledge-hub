function longestString(s) {
  let seen = {};
  let maxLen = 0;
  let start = 0;
  for (let i = 0; i < s.length; i++) {
    if (seen[s[i]] !== undefined && seen[s[i]] >= start) {
      start = seen[s[i]] + 1;
    }
    seen[s[i]] = i;
    maxLen = Math.max(maxLen, i - start + 1);
  }
  return maxLen;
}
